# creating a list to read the twee_json.txt into

tweet_list=[]
with open('tweet_json.txt') as f:
    for line in f:
        tweet_id=json.loads(line)['id']
        favorite_count=json.loads(line)['favorite_count']
        retweet_count=json.loads(line)['retweet_count']
        
        tweet_list.append({'tweet_id':tweet_id,
                         'favorite_count': favorite_count,
                         'retweet_count': retweet_count})

# Convert the Dictionary to DataFrame
tweet_df=pd.DataFrame(tweet_list, columns=['tweet_id', 'favorite_count', 'retweet_count'])
